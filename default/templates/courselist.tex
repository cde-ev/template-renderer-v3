<<# List of all courses to be used in the orga office or at the blackboard. With shortname, course instructors, course
    room and participant count for each track.

    Required template variables: None.
#>>
<<% extends "lists.base.tex" %>>

<<% set title = "Kursliste " + EVENT.title|e %>>
<<% set num_columns = 3
                      + (1 if EVENT.course_room_field else 0)
                      + EVENT.tracks|length %>>
<<% set hide_header = True %>>

<<% block table_head %>>
    \head{Nr.} & \head{Name}
    <<% if EVENT.course_room_field %>>
        & \head{Kursraum}
    <<% endif %>>
    & \head{Kursleiter}
    <<% for track in EVENT.tracks -%>>
        & \head{Teilnehmer<<% if EVENT.tracks|length > 1 %>> <<< track.shortname|e >>><<% endif %>>}
    <<% endfor %>>
<<% endblock %>>

<<% block table_body %>>
    <<% for course in EVENT.courses if course.is_active %>>
        <<< course.nr|e >>>. & <<< course.shortname|e >>>
        <<% if EVENT.course_room_field %>>
            & <<< course.fields[EVENT.course_room_field]|e >>>
        <<% endif %>>
        &
        <<% for instructor in course.instructors -%>>
            <<% if not loop.first %>>, <<% endif -%>>
            <<< instructor.name.given_names|e >>> <<< instructor.name.family_name|e >>>
        <<%- endfor %>>
        <<% for track in EVENT.tracks -%>>
            & <<% if course.tracks[track].status.is_active %>><<< course.tracks[track].attendees|length >>><<% endif %>>
        <<% endfor %>> \\
        \hline
    <<% endfor %>>
<<% endblock %>>
