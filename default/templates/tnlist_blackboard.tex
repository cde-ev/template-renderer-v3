<<% extends "lists.base.tex" %>>

<<% if not parts %>>
    <<% set parts = EVENT.parts %>>
<<% endif %>>
<<% if not tracks %>>
    <<% set tracks = UTIL.get_tracks(EVENT, parts=parts) %>>
<<% endif %>>
<<% if not registrations %>>
    <<% set stati = [ENUMS.RegistrationPartStati.participant] %>>
    <<% if CONFIG.tnlist.show_guests %>>
        <<% do stati.append(ENUMS.RegistrationPartStati.guest) %>>
    <<% endif %>>
    <<% set registrations = UTIL.get_registrations(EVENT, parts=parts, stati=stati, list_consent_only=False) %>>
<<% endif %>>
<<% set title = "Teilnehmerliste " + EVENT.title|e %>>
<<% if parts|length != EVENT.parts|length %>>
    <<% for part in parts %>>
        <<% set title = title + " " + part.title|e %>>
    <<% endfor %>>
<<% endif %>>
<<% set num_columns = 2 + EVENT.tracks|length + EVENT.parts|length %>>
<<% set portrait = (EVENT.tracks|length + EVENT.parts|length) < 3 %>>

<<% block table_head %>>
    \head{Vorname} & \head{Nachname}
    <<% for part in parts -%>>
        & \head{Unterkunft<<% if EVENT.parts|length > 1 %>> <<< part.shortname|e >>><<% endif %>>}
    <<% endfor %>>
    <<% for track in tracks -%>>
        & \head{Kurs<<% if EVENT.tracks|length > 1 %>> <<< track.shortname|e >>><<% endif %>>}
    <<% endfor %>>
<<% endblock %>>

<<% block table_body %>>
    <<% for registration in registrations %>>
        <<< registration.name.given_names|e >>> & <<< registration.name.family_name|e >>>
        <<% for part in EVENT.parts -%>>
            ~&
            <<# This only displays the guest flag, filtering should happen above. #>>
            <<% with status = registration.parts[part].status -%>>
                <<% if status == ENUMS.RegistrationPartStati.guest -%>>
                    \custombox{Gast}
                <<%- endif %>>
            <<%- endwith %>>
            <<% with lodgement = registration.parts[part].lodgement -%>>
                <<% if lodgement %>><<< lodgement.moniker|e >>><<% endif %>>
            <<%- endwith %>>
        <<% endfor %>>
        <<% for track in tracks -%>>
            & <<% if track in registration.tracks -%>>
                <<% with course = registration.tracks[track].course -%>>
                    <<% if course %>><<< course.nr|e >>>. <<< course.shortname|e >>><<% endif %>>
                <<% endwith -%>>
            <<% endif -%>>
        <<% endfor %>> \\
        \hline
    <<% endfor %>>
<<% endblock %>>
